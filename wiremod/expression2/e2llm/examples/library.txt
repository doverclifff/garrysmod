#[

Digital Screen internals:

Memory mapping [1]:
$[0, 786432) --------> Image data
$[786432, 1048569) --> unused [needs confirmation]
$[1048569, 1048576] -> Control registers, 8 in total

Control registers [1]:

There are 8 control registers.

$1048569: rw: Color mode
$1048570: rw: Clear row (Y)
$1048571: rw: Clear column (X)
$1048572: rw: Vertical resolution (Y) in pixels
$1048573: rw: Horizontal resolution (X) in pixels
$1048574: rw: Hardware clear screen
$1048575: rw: Clk
$1048576: rw: Compression enable

Color modes [1][needs clarification]:

There are four color modes. They can be selected by
writing magic values to the color mode control register.
There are four magic values, each corresponding to a different
color mode.

0: RGBXXX
1: R G B
2: 65536 * R + 256 * G + 1 * B
3: RRRGGGBBB

The function rgb2digi(V, N) -> N provides an easy
way to map RGB-encoded color data to the digital screen color space.

Clk behavior:

Digital screen clocking is slightly unintuitive. When the Clk
control register is set 1 (or True), any following writes to the screen memory
take effect immediately / as fast as possible. When Clk equals 0 (or False),
writes are buffered until the next rising-edge of Clk.

This contrasts with the behavior of the Console Screen, which permits only a
single write per clock rising-edge.

Clear screen register behavior:

Writing 0 to the hardware clear register ($1048574 = 0) will trigger a screen clear. [1]

Sources:
[1] https://web.archive.org/web/20160323170932/http://wiki.wiremod.com/wiki/Digital_Screen

]#

# @doc Raises an unrecoverable error
# @name dsError
# @out NoReturn
function void dsError(Caller:string, Msg:string) {
    let DSLibSemVerString = DRV_DIGITALSCREEN_VERSION:semVerGetString() + ": "
    let ErrString = DSLibSemVerString + "in " + Caller + ": " + Msg
    error(ErrString)
}

# @doc Raises an unrecoverable error IFF the passed expression is falsy.
# @name dsAssert
# @out Number | NoReturn
function number dsAssert(Expression:number, Caller:string, Msg:string) {
    if (!Expression) {
        dsError(Caller, DS_STRINGS["error_assertion_violated", string] + " " + Msg)
    }
    return 1
}

# @doc   Guarantees that the passed register name is contained in the set DS_REGISTER_FILE_MAP:keys(). Returns 1. Else, raises an error.
# @name  dsValidateRegisterNameOrError
# @group Primitive, Validation
# @out   Number
function number dsValidateRegisterNameOrError(RegisterName:string) {
    if (DS_REGISTER_FILE_MAP:keys():indexOf(RegisterName) == 0) {
        dsError(
            "dsValidateRegisterNameOrError",
            DS_STRINGS["error_invalid_register_name", string]
        )
    }
    return 1
}

# @doc   Guarantees that the coordinate pair XY is within the screen's bounds, returns 1. Else, raises an error.
# @name  dsValidateXYIsInScreenBounds
# @group Primitive, Validation
# @out   Number
function number wirelink:dsValidateXYIsInScreenBounds(XY:vector2) {
    let ResolutionX = This[DS_REGISTER_FILE_MAP["ResolutionX", number]]
    let ResolutionY = This[DS_REGISTER_FILE_MAP["ResolutionY", number]]
    let X = XY:x()
    let Y = XY:y()
    let Caller = "dsValidateXYIsInScreenBounds"
    let ErrString = DS_STRINGS["error_vec2_out_of_bounds_screenspace", string]
    dsAssert(
        X >= 0, 
        Caller, 
        format(ErrString, "X", X, Y, "to the left of", ResolutionX, ResolutionY)
    )
    dsAssert(
        X < ResolutionX, 
        Caller, 
        format(ErrString, "X", X, Y, "to the right of", ResolutionX, ResolutionY)
    )
    dsAssert(
        Y >= 0, 
        Caller, 
        format(ErrString, "Y", X, Y, "above the", ResolutionX, ResolutionY)
    )
    dsAssert(
        Y < ResolutionY, 
        Caller, 
        format(ErrString, "Y", X, Y, "below the", ResolutionX, ResolutionY)
    )
    return 1
}

# @doc   Guarantees that the address pointed to by Ptr is within the bounds of the specified memory extent, returns 1. Else, raises an error.
# @name  dsValidatePointsToExtentOrError
# @group Primitive, Validation
# @out   Number
function number dsValidatePointsToExtentOrError(Ptr:number, ExtentName:string) {
    let Caller = "dsValidatePointsToExtentOrError"
    let ErrString = DS_STRINGS["error_invalid_extent_name", string]
    dsAssert(
        DS_MEMORY_EXTENT_NAMES:indexOf(ExtentName) != 0,
        Caller,
        format(ErrString, ExtentName)
    )
    let ExtentStart = 0
    let ExtentEnd = 0
    if ( ExtentName == "ImageData" ) {
        ExtentStart = dsGetOffsetStartImageData()
        ExtentEnd = dsGetOffsetEndImageData()
    } elseif ( ExtentName == "Unallocated" ) {
        ExtentStart = dsGetOffsetStartUnallocated()
        ExtentEnd = dsGetOffsetEndUnallocated()
    } elseif ( ExtentName == "RegisterFile" ) {
        ExtentStart = dsGetOffsetStartRegisterFile()
        ExtentEnd = dsGetOffsetEndRegisterFile()
    }
    
    ErrString = DS_STRINGS["error_address_not_in_extent", string]
    dsAssert(
        Ptr >= ExtentStart, 
        Caller, 
        format(ErrString, Ptr, ExtentName, ExtentStart, ExtentEnd)
    )
    dsAssert(
        Ptr <= ExtentEnd, 
        Caller, 
        format(ErrString, Ptr, ExtentName, ExtentStart, ExtentEnd)
    )
    return 1
}

# @doc   Guarantees that the address pointed to by Ptr is within the bounds of the ImageData extent, returns 1. Else, raises an error.
# @name  dsValidatePointsToImageDataExtentOrError
# @group Primitive, Validation
# @out   Number
function number dsValidatePointsToImageDataExtentOrError(Ptr:number) {
    let ExtentStart = dsGetOffsetStartImageData()
    let ExtentEnd = dsGetOffsetEndImageData()
    
    let Caller = "dsValidatePointsToImageDataExtentOrError"
    let ErrString = "Address %d is outside of the ImageData memory extent [%d, %d]."
    dsAssert(
        Ptr >= ExtentStart, 
        Caller, 
        format(ErrString, Ptr, ExtentStart, ExtentEnd)
    )
    dsAssert(
        Ptr <= ExtentEnd, 
        Caller, 
        format(ErrString, Ptr, ExtentStart, ExtentEnd)
    )
    return 1
}

# @name  dsWriteClkEnableToggle
# @doc   Reads the value stored in the ClkEnable register, inverts it, then overwrites ClkEnable with the new value. Returns the new value of ClkEnable.
# @out   Number
# @group Helper, Write (hispeed)
function number wirelink:dsWriteClkEnableToggle() {
    return This:dsWriteClkEnable(!This:dsReadClkEnable())
}

# @doc   Overwrites ALL named registers with sane default values.
# @name  dsWriteResetAllRegisters
# @out   Number
# @group Helper, Write (hispeed)
function void wirelink:dsWriteResetAllRegisters() {
    foreach ( RegisterName:string, _:number = DS_REGISTER_DEFAULT_VALUE ) {
        This:dsWriteRegisterDefaultValue(RegisterName)
    }
    return 1
}
