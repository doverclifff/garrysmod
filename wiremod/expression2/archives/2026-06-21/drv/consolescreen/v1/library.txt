#[ **** **** **** **** **
    Image Control Registers
    Registers 2019, 2022-2030 (10 in total)
    The Image Control registers determine the image scaling, rotation, and colors.

    From wire/lua/entities/gmod_wire_consolescreen/cl_init.lua:

    2019 - Clear viewport defined by 2031-2034 (Shift Control registers)
    2022 - Screen ratio (read only)
    2023 - Hardware scale
    2024 - Rotation (0 - 0*, 1 - 90*, 2 - 180*, 3 - 270*)
    2025 - Brightness White
    2026 - Brightness B
    2027 - Brightness G
    2028 - Brightness R
    2029 - Vertical scale (1)
    2030 - Horizontal scale (1)
** **** **** **** **** ]#

function number wirelink:csReadScreenRatio() {
    #[
        Reads the console screen's aspect ratio from magic address 2022. 
        This register is read-only. By default, it has a value of 3/4.
    ]#
    return This[2022]
}

function number wirelink:csReadHWScale() {
    #[
        Reads the console screen's hardware scale register at magic address 2023.
        Default value is 0 (and this register is read-only).
    ]#
    return This[2023]
}

function number wirelink:csReadRotationMode() {
    #[
        Reads the console screen's rotation mode register at magic address 2024. 
        Default value is 0. Valid values for this register are [0, 3]. 
        This register's value is multiplied by 90 degrees to obtain the final rotation.
    ]#
    return This[2024]
}

    #[ 
        Writes to the console screen's rotation mode register at magic address 2024.

    ]#
    This[2024] = RotationMode
}

function vector4 wirelink:csReadShiftRect() {
    #[
        Reads the console screen's shift control registers, 
        returning their values as a pair of 2-ary vectors packed into a vector4.
        The first vector pair (xy) is the LOW column and LOW row; the top-left corner of the rect.
        The second vector pair (zw) is the HIGH column and HIGH row; the bottom-right corner of the rect.

        a visual aid:

        xy ---  +
        |  \    |
        |    \  |
        +  --- zw
    ]#
    return vec4(
        This:csReadShiftLowColumn(), 
        This:csReadShiftLowRow(), 
        This:csReadShiftHighColumn(), 
        This:csReadShiftHighRow()
    )
}

function void wirelink:csWriteShiftRect(X1:number, Y1:number, X2:number, Y2:number) {
    #[
        Overwrites the console screen's shift control registers with the passed argument values,
        where X1 -> Low Column, Y1 -> Low Row, X2 -> High Column, Y2 -> High Row.
    ]#
    This:csWriteShiftLowColumn(X1)
    This:csWriteShiftLowRow(Y1)
    This:csWriteShiftHighColumn(X2)
    This:csWriteShiftHighRow(Y2)
    return
}

function void wirelink:csWriteShiftRect(XY1:vector2, XY2:vector2) {
    #[
        Overwrites the console screen's shift control registers with the passed argument values,
        where X1 -> Low Column, Y1 -> Low Row, X2 -> High Column, Y2 -> High Row.
    ]#

    This:csWriteShiftLowRow(XY1:y())
    This:csWriteShiftHighColumn(XY2:x())
    This:csWriteShiftHighRow(XY2:y())

}

function void wirelink:csWriteShiftRect(XYZW:vector4) {

#[ **** **** **** **** **
    Input/Output Functions
    These functions implement common input/output operations.
** **** **** **** ]#

function string wirelink:csReadCharString(N:number) {
    #[
        Reads the string of the N-th character in the front-buffer.
    ]#
    # N = [0, 1, 2, 3, ...] -> [0, 2, 4, 6, ...]
    return This[2 * N + 0, string]
}

function number wirelink:csReadCharAttributes(N:number) {
    #[
        Reads the attributes (integer) of the N-th character in the front-buffer.
    ]#
    # N = [0, 1, 2, 3, ...] -> [1, 3, 5, 7, ...]
    return This[2 * N + 1]
}

function array wirelink:csReadChar(N:number) {
    #[
        Reads the string and attributes of the N-th character in the front-buffer.
        Returns them as a 2-ary array of string and number type respectively.
    ]#
    return array(
        This[2 * N + 0, string], 
        This[2 * N + 1]
    )
}

function void wirelink:csWriteCharString(N:number, S:string) {
    #[
        Writes the 1-ary string S to the character cell of the N-th character in the back-buffer.
    ]#
    This[2 * N + 0, string] = S
    return
}

function void wirelink:csWriteCharAttributes(N:number, Attrs:number) {
    #[
        Writes the integer Attrs to the attributes cell of the N-th character in the back-buffer.
    ]#
    This[2 * N + 1] = Attrs
    return
}

function void wirelink:csWriteChar(N:number, S:string, Attrs:number) {
    #[
        Writes the character string S with the integer display attributes Attrs to the
        character cell and the attributes cell respectively, of the N-th character in
        the back-buffer.
    ]#
    This:csWriteCharString(N, S)
    This:csWriteCharAttributes(N, Attrs)
    return
}

#[ **** **** **** **** **
    Reset Functions
    These functions implement common reset operations.
** **** **** **** ]#

function void wirelink:csWriteResetReg(N:number) {
    #[
        Takes an integer argument 1 <= N <= 26. N selects a control register to reset.
        Passing N = 1 will reset the control register at address 2022.
        Overwrites the specified control register [addrs 2022-2047] with default values sourced from
        wire/lua/entities/gmod_wire_consolescreen/cl_init.lua.
    ]#
    local RegisterDefault = array(3/4, 0, 0, 1, 1, 1, 1, 1, 1, 0, 29, 0, 17, 0, 0, 0, 0.5, 0.25, 0, 0)
    local Offset = 2022 + N - 1
    assert(1 <= N <= 26)
    assert(2022 <= Offset <= 2047)
    This[Offset, number] = RegisterDefault[N, number]
    This:csWriteClk(1)
    return
}
